Chapter 1: C# Basics

What is C#?

C# (pronounced "C Sharp") is a modern, object-oriented programming language developed by Microsoft. It's part of the .NET ecosystem and is widely used for Windows applications, web development, and game development with Unity.

Key Features

  • Object-Oriented: Full support for OOP principles
  • Type-Safe: Compile-time type checking
  • Modern Syntax: Clean and readable code
  • Cross-Platform: Runs on Windows, Linux, macOS
  • Rich Ecosystem: Extensive .NET libraries and frameworks

Variables in C#

C# is strongly typed - variables must be declared with specific types:

Variable Declaration

Chapter 3: Control Structures

Conditional Statements

Control program flow with if-else and switch statements:

If-Else Statements

using System;

class ConditionalExample
{
    static void Main()
    {
        int age = 20;
        
        if (age >= 18)
        {
            Console.WriteLine("You are an adult");
        }
        else if (age >= 13)
        {
            Console.WriteLine("You are a teenager");
        }
        else
        {
            Console.WriteLine("You are a child");
        }
        
        // Switch statement
        string day = "Monday";
        
        switch (day)
        {
            case "Monday":
                Console.WriteLine("Start of work week");
                break;
            case "Friday":
                Console.WriteLine("Weekend is coming!");
                break;
            default:
                Console.WriteLine("Regular day");
                break;
        }
    }
}

Loops

Repeat actions with for, while, and foreach loops:

Loop Examples

using System;

class LoopExample
{
    static void Main()
    {
        // For loop
        Console.WriteLine("For loop:");
        for (int i = 0; i < 5; i++)
        {
            Console.WriteLine("Number: " + i);
        }
        
        // While loop
        Console.WriteLine("While loop:");
        int count = 0;
        while (count < 3)
        {
            Console.WriteLine("Count: " + count);
            count++;
        }
        
        // Foreach loop (for collections)
        Console.WriteLine("Foreach loop:");
        string[] fruits = { "apple", "banana", "orange" };
        foreach (string fruit in fruits)
        {
            Console.WriteLine(fruit);
        }
        
        // Do-while loop
        Console.WriteLine("Do-while loop:");
        int x = 0;
        do
        {
            Console.WriteLine("x = " + x);
            x++;
        } while (x < 3);
    }
}

Chapter 4: Methods and Classes

Methods (Functions)

Reusable blocks of code in C#:

Method Examples

using System;

class MethodExample
{
    // Simple method
    static void Greet()
    {
        Console.WriteLine("Hello, C#!");
    }
    
    // Method with parameters and return type
    static int Add(int a, int b)
    {
        return a + b;
    }
    
    // Method with default parameter
    static double CalculateArea(double radius, double pi = 3.14159)
    {
        return pi * radius * radius;
    }
    
    // Method overloading
    static string CombineNames(string firstName, string lastName)
    {
        return firstName + " " + lastName;
    }
    
    static string CombineNames(string firstName, string middleName, string lastName)
    {
        return firstName + " " + middleName + " " + lastName;
    }
    
    static void Main()
    {
        Greet();  // Call method without parameters
        
        int result1 = Add(5, 3);      // 8
        double result2 = CalculateArea(5); // 78.53975
        
        Console.WriteLine("Addition: " + result1);
        Console.WriteLine("Area: " + result2);
        
        string fullName1 = CombineNames("John", "Doe");
        string fullName2 = CombineNames("John", "Michael", "Doe");
    }
}

Classes and Objects

Object-Oriented Programming in C#:

Class and Object Example

using System;

// Person class definition
public class Person
{
    // Fields (attributes)
    private string name;
    private int age;
    
    // Properties (with getters and setters)
    public string Name
    {
        get { return name; }
        set { name = value; }
    }
    
    public int Age
    {
        get { return age; }
        set { age = value; }
    }
    
    // Constructor
    public Person(string name, int age)
    {
        Name = name;
        Age = age;
    }
    
    // Method
    public void Introduce()
    {
        Console.WriteLine($"Hello, my name is {Name} and I am {Age} years old.");
    }
}

// Main program
class Program
{
    static void Main()
    {
        // Create objects
        Person person1 = new Person("Alice", 25);
        Person person2 = new Person("Bob", 30);
        
        // Use object methods
        person1.Introduce();
        person2.Introduce();
        
        // Use properties
        person1.Name = "Alicia";
        Console.WriteLine("New name: " + person1.Name);
    }
}

Chapter 5: Advanced C# Features

Collections and LINQ

Working with data collections and Language Integrated Query:

Collections and LINQ Examples

using System;
using System.Collections.Generic;
using System.Linq;

class CollectionsExample
{
    static void Main()
    {
        // Arrays
        int[] numbers = { 1, 2, 3, 4, 5 };
        
        // Lists (dynamic arrays)
        List<string> fruits = new List<string>();
        fruits.Add("apple");
        fruits.Add("banana");
        fruits.Add("orange");
        
        // Dictionaries (key-value pairs)
        Dictionary<string, int> ages = new Dictionary<string, int>();
        ages.Add("Alice", 25);
        ages.Add("Bob", 30);
        
        // LINQ queries
        var evenNumbers = from n in numbers
                           where n % 2 == 0
                           select n;
        
        var longFruits = fruits.Where(f => f.Length > 5);
        
        // Working with collections
        Console.WriteLine("Even numbers:");
        foreach (var num in evenNumbers)
        {
            Console.WriteLine(num);
        }
        
        Console.WriteLine("Bob's age: " + ages["Bob"]);
    }
}

Exception Handling

Handling errors and exceptions in C#:

Exception Handling

using System;

class ExceptionExample
{
    static void Main()
    {
        try
        {
            Console.Write("Enter a number: ");
            string input = Console.ReadLine();
            int number = int.Parse(input);
            
            int result = 100 / number;
            Console.WriteLine("Result: " + result);
        }
        catch (FormatException)
        {
            Console.WriteLine("Error: Please enter a valid number.");
        }
        catch (DivideByZeroException)
        {
            Console.WriteLine("Error: Cannot divide by zero.");
        }
        catch (Exception ex)
        {
            Console.WriteLine($"An error occurred: {ex.Message}");
        }
        finally
        {
            Console.WriteLine("Program execution completed.");
        }
    }
}

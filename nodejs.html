Chapter 1: Node.js Basics

What is Node.js?

Node.js is a JavaScript runtime built on Chrome's V8 JavaScript engine that allows you to run JavaScript on the server-side.

Key Features

  • Non-blocking I/O: Asynchronous operations
  • Event-driven: Uses event loop for handling requests
  • Single-threaded: But highly scalable with event loop
  • Cross-platform: Runs on Windows, Linux, macOS
  • NPM: Largest package ecosystem

Installing Node.js

Download and install Node.js from the official website or use version managers:

Using NVM (Node Version Manager)

# Install NVM
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.0/install.sh | bash

# Install latest Node.js
nvm install node

# Use specific version
nvm install 18.0.0
nvm use 18.0.0

Verifying Installation

# Check Node.js version
node --version

# Check npm version
npm --version

# Run JavaScript file
node app.js

Chapter 2: Modules System

CommonJS Modules

Node.js uses CommonJS module system for organizing code:

Exporting Modules

// math.js - Exporting functions
const add = (a, b) => a + b;
const subtract = (a, b) => a - b;
const multiply = (a, b) => a * b;

// Export as object
module.exports = {
  add,
  subtract,
  multiply
};

// Or export individually
exports.add = add;
exports.PI = 3.14159;

Importing Modules

// app.js - Importing modules
const math = require('./math.js');
const fs = require('fs'); // Built-in module

console.log(math.add(5, 3)); // 8
console.log(math.PI); // 3.14159

// Destructuring imports
const { add, multiply } = require('./math.js');

ES6 Modules (Alternative)

Node.js also supports ES6 module syntax:

ES6 Module Syntax

// math.mjs (or package.json with "type": "module")
export const add = (a, b) => a + b;
export const PI = 3.14159;

export default function multiply(a, b) {
  return a * b;
}

// Importing
import { add, PI } from './math.mjs';
import multiply from './math.mjs';

Chapter 3: HTTP Server

Creating HTTP Server

Node.js has a built-in HTTP module to create web servers:

Basic HTTP Server

const http = require('http');

// Create server
const server = http.createServer((req, res) => {
  // Set response header
  res.setHeader('Content-Type', 'text/html');
  
  // Handle different routes
  if (req.url === '/') {
    res.write('<h1>Welcome to Home Page</h1>');
  } else if (req.url === '/about') {
    res.write('<h1>About Us</h1>');
  } else {
    res.write('<h1>Page Not Found</h1>');
  }
  
  res.end();
});

// Start server
server.listen(3000, () => {
  console.log('Server running on http://localhost:3000');
});

Handling JSON Responses

Creating REST API endpoints:

JSON API Server

const http = require('http');

const server = http.createServer((req, res) => {
  // Set CORS headers
  res.setHeader('Access-Control-Allow-Origin', '*');
  res.setHeader('Content-Type', 'application/json');

  if (req.url === '/api/users' && req.method === 'GET') {
    const users = [
      { id: 1, name: 'John Doe', email: 'john@example.com' },
      { id: 2, name: 'Jane Smith', email: 'jane@example.com' }
    ];
    
    res.end(JSON.stringify(users));
  } else {
    res.end(JSON.stringify({ error: 'Route not found' }));
  }
});

server.listen(3000, () => {
  console.log('JSON API running on port 3000');
});

Chapter 4: File System

File Operations

Node.js provides the fs module for file system operations:

File System Methods

const fs = require('fs');
const path = require('path');

// Reading files
fs.readFile('./file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});

// Writing files
fs.writeFile('./output.txt', 'Hello World!', (err) => {
  if (err) throw err;
  console.log('File written successfully');
});

// Promises version (fs/promises)
const fsPromise = require('fs').promises;

async function readFileAsync() {
  try {
    const data = await fsPromise.readFile('./file.txt', 'utf8');
    console.log(data);
  } catch (err) {
    console.error(err);
  }
}

Working with Directories

Managing folders and file paths:

Directory Operations

const fs = require('fs');

// Create directory
fs.mkdir('./new-folder', { recursive: true }, (err) => {
  if (err) throw err;
  console.log('Directory created');
});

// Read directory contents
fs.readdir('./', (err, files) => {
  if (err) throw err;
  files.forEach(file => {
    console.log(file);
  });
});

// Check if file/directory exists
if (fs.existsSync('./file.txt')) {
  console.log('File exists');
}

Chapter 5: NPM and Packages

Package.json

The package.json file contains project metadata and dependencies:

package.json Structure

{
  "name": "my-node-app",
  "version": "1.0.0",
  "description": "A simple Node.js application",
  "main": "app.js",
  "scripts": {
    "start": "node app.js",
    "dev": "nodemon app.js",
    "test": "jest"
  },
  "dependencies": {
    "express": "^4.18.0",
    "mongoose": "^6.0.0"
  },
  "devDependencies": {
    "nodemon": "^2.0.0",
    "jest": "^27.0.0"
  }
}

NPM Commands

Essential npm commands for package management:

Common NPM Commands

npm install -g nodemon

# Run scripts
npm start
npm run dev
npm test

# Update packages

npm outdated # check for updates

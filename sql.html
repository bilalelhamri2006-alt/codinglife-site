Chapter 1: SQL Basics

What is SQL?

SQL (Structured Query Language) is a standard language for managing and manipulating relational databases. It's used to store, retrieve, and manipulate data in databases.

Key Features

  • Standard Language: Works with most database systems
  • Declarative: You specify what you want, not how to get it
  • Powerful: Handles complex queries and data operations
  • Scalable: Works with small to enterprise-level databases
  • Secure: Provides data access control and permissions

Chapter 2: Database Design & Tables

Creating Databases and Tables

Fundamental operations for database structure:

Database and Table Operations

-- Create a database
CREATE DATABASE company;

-- Use the database
USE company;

-- Create a table
CREATE TABLE employees (
    id INT PRIMARY KEY AUTO_INCREMENT,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    email VARCHAR(100) UNIQUE,
    salary DECIMAL(10, 2),
    hire_date DATE,
    department_id INT
);

-- Create departments table
CREATE TABLE departments (
    department_id INT PRIMARY KEY AUTO_INCREMENT,
    department_name VARCHAR(100) NOT NULL,
    manager_id INT
);

Chapter 3: CRUD Operations

INSERT - Adding Data

Insert new records into tables:

INSERT Examples

-- Insert single record
INSERT INTO employees (first_name, last_name, email, salary, hire_date)
VALUES ('John', 'Doe', 'john.doe@company.com', 75000, '2023-01-15');

-- Insert multiple records
INSERT INTO employees (first_name, last_name, email, salary)
VALUES 
    ('Jane', 'Smith', 'jane.smith@company.com', 68000),
    ('Bob', 'Johnson', 'bob.johnson@company.com', 72000),
    ('Alice', 'Brown', 'alice.brown@company.com', 81000);

-- Insert with SELECT (copy data)
INSERT INTO employee_archive
SELECT * FROM employees 
WHERE hire_date < '2020-01-01';

SELECT - Retrieving Data

Query data from tables:

SELECT Examples

-- Select all columns
SELECT * FROM employees;

-- Select specific columns
SELECT first_name, last_name, salary 
FROM employees;

-- Select with aliases
SELECT 
    first_name AS 'First Name',
    last_name AS 'Last Name',
    salary AS 'Annual Salary'
FROM employees;

-- Select with calculated columns
SELECT 
    first_name,
    last_name,
    salary,
    salary / 12 AS 'Monthly Salary'
FROM employees;

UPDATE and DELETE

Modify and remove data:

UPDATE and DELETE Examples

-- UPDATE specific records
UPDATE employees
SET salary = salary * 1.1  -- 10% raise
WHERE department_id = 2;

-- UPDATE multiple columns
UPDATE employees
SET 
    salary = 80000,
    email = 'new.email@company.com'
WHERE id = 5;

-- DELETE specific records
DELETE FROM employees
WHERE hire_date < '2010-01-01';

-- DELETE all records (be careful!)

Chapter 4: Filtering and Sorting

WHERE Clause

Filter records based on conditions:

WHERE Clause Examples

-- Basic comparison operators
SELECT * FROM employees 
WHERE salary > 70000;

SELECT * FROM employees 
WHERE department_id = 3;

-- String matching with LIKE
SELECT * FROM employees 
WHERE last_name LIKE 'S%';  -- Starts with S

SELECT * FROM employees 
WHERE email LIKE '%@gmail.com';  -- Ends with @gmail.com

-- Date range filtering
SELECT * FROM employees 
WHERE hire_date BETWEEN '2023-01-01' AND '2023-06-30';

-- Multiple conditions with AND/OR
SELECT * FROM employees 
WHERE salary > 60000 
  AND department_id = 2;

SELECT * FROM employees 
WHERE department_id = 1 
  OR department_id = 3;

ORDER BY and LIMIT

Sort and limit query results:

Sorting and Limiting

-- Sort by single column
SELECT first_name, last_name, salary
FROM employees
ORDER BY salary DESC;  -- Highest salary first

-- Sort by multiple columns
SELECT department_id, last_name, first_name
FROM employees
ORDER BY department_id ASC, last_name ASC;

-- Limit number of results
SELECT * FROM employees
ORDER BY salary DESC
LIMIT 5;  -- Top 5 highest paid

-- Limit with offset (pagination)
SELECT * FROM employees
ORDER BY id
LIMIT 10 OFFSET 20;  -- Rows 21-30
WHERE LIKE BETWEEN ORDER BY LIMIT

Chapter 5: Advanced Queries

JOIN Operations

Combine data from multiple tables:

JOIN Examples

-- INNER JOIN (most common)
SELECT 
    e.first_name,
    e.last_name,
    e.salary,
    d.department_name
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id;

-- LEFT JOIN (all employees, even without department)
SELECT 
    e.first_name,
    e.last_name,
    d.department_name
FROM employees e
LEFT JOIN departments d ON e.department_id = d.department_id;

-- Multiple JOINs
SELECT 
    e.first_name,
    e.last_name,
    d.department_name,
    m.first_name AS 'manager_first_name',
    m.last_name AS 'manager_last_name'
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id
LEFT JOIN employees m ON d.manager_id = m.id;

Aggregate Functions and GROUP BY

Perform calculations on groups of data:

Aggregate Functions

-- Basic aggregate functions
SELECT 
    COUNT(*) AS 'total_employees',
    AVG(salary) AS 'average_salary',
    MAX(salary) AS 'max_salary',
    MIN(salary) AS 'min_salary',
    SUM(salary) AS 'total_salary_budget'
FROM employees;

-- GROUP BY with departments
SELECT 
    d.department_name,
    COUNT(e.id) AS 'employee_count',
    AVG(e.salary) AS 'avg_salary',
    MAX(e.salary) AS 'max_salary'
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id
GROUP BY d.department_name
ORDER BY 'avg_salary' DESC;

-- HAVING clause (filter grouped results)
SELECT 
    d.department_name,
    AVG(e.salary) AS 'avg_salary'
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id
GROUP BY d.department_name
HAVING AVG(e.salary) > 70000;
